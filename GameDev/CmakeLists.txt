#this is a function which adds .dll files to project path
function(CopyLibToTarget LIB_NAME TARGET_NAME)
	add_custom_command(
		TARGET ${TARGET_NAME}
		POST_BUILD
		COMMAND
		${CMAKE_COMMAND} -E copy_if_different
		$<TARGET_FILE:${LIB_NAME}>
		$<TARGET_FILE_DIR:${TARGET_NAME}>
	)
endfunction()

CopyLibToTarget(sfml-window ${Game_Name_Target})
CopyLibToTarget(sfml-graphics ${Game_Name_Target})
CopyLibToTarget(sfml-system ${Game_Name_Target})
CopyLibToTarget(sfml-audio ${Game_Name_Target})
