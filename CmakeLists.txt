cmake_minimum_required(VERSION 4.0)
project(GameDevelopment
	VERSION 1.0.0
	LANGUAGES C CXX
)

set(CXX_STANDARD			14)
set(CXX_STANDARD_REQUIRED	ON)
set(CXX_EXTENSIONS		   OFF)

#this creates a variable
set(Game_Name_Target GamePlay)

add_subdirectory(GameDev)

# this is fetching sfml library
include(FetchContent)
set(SFML_LIB_NAME SFML)
FetchContent_Declare(
	${SFML_LIB_NAME}
	GIT_REPOSITORY https://github.com/SFML/SFML
	GIT_TAG 2.6.2
	GIT_SHALLOW FALSE
)

FetchContent_MakeAvailable(${SFML_LIB_NAME})

# link sfml library to the project
target_link_libraries(${Game_Name_Target} PUBLIC sfml-graphics)
target_link_libraries(${Game_Name_Target} PUBLIC sfml-window)
target_link_libraries(${Game_Name_Target} PUBLIC sfml-system)
target_link_libraries(${Game_Name_Target} PUBLIC sfml-audio)
